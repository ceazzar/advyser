## API Endpoints

### Get Claim Status
```
GET /api/v1/claims/:claimId
Response: {
  id: string,
  reference: string,
  businessId: string,
  businessName: string,
  abn: string,
  advisorType: string,
  status: "submitted" | "under_review" | "action_required" |
          "awaiting_approval" | "approved" | "denied",
  submittedAt: timestamp,
  updatedAt: timestamp,
  estimatedReviewDate: date,
  verificationChecks: [
    { name: string, status: "pending" | "in_progress" | "complete" }
  ],
  role: "owner" | "staff",
  ownerEmail?: string,
  ownerApprovalStatus?: "pending" | "approved" | "denied",
  notificationEmail: string,
  additionalInfoRequest?: {
    requestedAt: timestamp,
    message: string,
    dueDate: date
  },
  denialReason?: string
}
```

### Submit Response
```
POST /api/v1/claims/:claimId/response
Body: {
  documents: [{ id, name, size }],
  comment: string
}
Response: {
  success: boolean,
  claimId: string,
  newStatus: "under_review"
}
```

### Resend Owner Approval
```
POST /api/v1/claims/:claimId/resend-approval
Response: {
  sent: boolean,
  email: string,
  sentAt: timestamp
}
```

### Cancel Claim
```
DELETE /api/v1/claims/:claimId
Response: {
  success: boolean,
  claimId: string
}
```

## State Management
```typescript
interface ClaimStatusState {
  claim: {
    id: string;
    reference: string;
    businessId: string;
    businessName: string;
    abn: string;
    advisorType: string;
    status: ClaimStatus;
    submittedAt: Date;
    updatedAt: Date;
    estimatedReviewDate: Date;
    verificationChecks: VerificationCheck[];
    role: 'owner' | 'staff';
    ownerEmail?: string;
    ownerApprovalStatus?: 'pending' | 'approved' | 'denied';
    notificationEmail: string;
    additionalInfoRequest?: AdditionalInfoRequest;
    denialReason?: string;
  };
  isLoading: boolean;
  error?: string;
  response: {
    documents: File[];
    comment: string;
    isSubmitting: boolean;
  };
}

type ClaimStatus =
  | 'submitted'
  | 'under_review'
  | 'action_required'
  | 'awaiting_approval'
  | 'approved'
  | 'denied';

interface VerificationCheck {
  name: string;
  status: 'pending' | 'in_progress' | 'complete';
}

interface AdditionalInfoRequest {
  requestedAt: Date;
  message: string;
  dueDate: Date;
}
```
